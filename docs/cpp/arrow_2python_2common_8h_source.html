36 class ARROW_EXPORT PyAcquireGIL {
37  public:
38  PyAcquireGIL() : acquired_gil_(false) { acquire(); }
39 
40  ~PyAcquireGIL() { release(); }
41 
42  void acquire() {
43  if (!acquired_gil_) {
44  state_ = PyGILState_Ensure();
45  acquired_gil_ = true;
46  }
47  }
48 
49  // idempotent
50  void release() {
51  if (acquired_gil_) {
52  PyGILState_Release(state_);
53  acquired_gil_ = false;
54  }
55  }
56 
57  private:
58  bool acquired_gil_;
59  PyGILState_STATE state_;
60  ARROW_DISALLOW_COPY_AND_ASSIGN(PyAcquireGIL);
61 };

113 struct ARROW_EXPORT PyObjectStringify {
114  OwnedRef tmp_obj;
115  const char* bytes;
116  Py_ssize_t size;
117 
118  explicit PyObjectStringify(PyObject* obj) {
119  PyObject* bytes_obj;
120  if (PyUnicode_Check(obj)) {
121  bytes_obj = PyUnicode_AsUTF8String(obj);
122  tmp_obj.reset(bytes_obj);
123  bytes = PyBytes_AsString(bytes_obj);
124  size = PyBytes_GET_SIZE(bytes_obj);
125  } else if (PyBytes_Check(obj)) {
126  bytes = PyBytes_AsString(obj);
127  size = PyBytes_GET_SIZE(obj);
128  } else {
129  bytes = NULLPTR;
130  size = -1;
131  }
132  }
133 };
