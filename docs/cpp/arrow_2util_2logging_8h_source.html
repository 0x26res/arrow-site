106 // To make the logging lib plugable with other logging libs and make
107 // the implementation unawared by the user, ArrowLog is only a declaration
108 // which hide the implementation into logging.cc file.
109 // In logging.cc, we can choose different log libs using different macros.

111 // This is also a null log which does not output anything.
112 class ARROW_EXPORT ArrowLogBase {
113  public:
114  virtual ~ArrowLogBase() {}
115 
116  virtual bool IsEnabled() const { return false; }
117 
118  template <typename T>
119  ArrowLogBase& operator<<(const T& t) {
120  if (IsEnabled()) {
121  Stream() << t;
122  }
123  return *this;
124  }
125 
126  protected:
127  virtual std::ostream& Stream() { return std::cerr; }
128 };
