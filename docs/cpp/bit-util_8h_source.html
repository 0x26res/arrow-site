105 inline int64_t RoundUpToMultipleOf64(int64_t num) {
106  // TODO(wesm): is this definitely needed?
107  // DCHECK_GE(num, 0);
108  constexpr int64_t round_to = 64;
109  constexpr int64_t force_carry_addend = round_to - 1;
110  constexpr int64_t truncate_bitmask = ~(round_to - 1);
111  constexpr int64_t max_roundable_num = std::numeric_limits<int64_t>::max() - round_to;
112  if (num <= max_roundable_num) { return (num + force_carry_addend) & truncate_bitmask; }
113  // handle overflow case. This should result in a malloc error upstream
114  return num;
115 }
