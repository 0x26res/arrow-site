174  Status Resize(int64_t elements) {
175  // Resize(0) is a no-op
176  if (elements == 0) { return Status::OK(); }
177  if (capacity_ == 0) { buffer_ = std::make_shared<PoolBuffer>(pool_); }
178  int64_t old_capacity = capacity_;
179  RETURN_NOT_OK(buffer_->Resize(elements));
180  capacity_ = buffer_->capacity();
181  data_ = buffer_->mutable_data();
182  if (capacity_ > old_capacity) {
183  memset(data_ + old_capacity, 0, capacity_ - old_capacity);
184  }
185  return Status::OK();
186  }
187 
188  Status Append(const uint8_t* data, int64_t length) {
189  if (capacity_ < length + size_) { RETURN_NOT_OK(Resize(length + size_)); }
190  UnsafeAppend(data, length);
191  return Status::OK();
192  }
193 
194  // Advance pointer and zero out memory
195  Status Advance(int64_t length) {
196  if (capacity_ < length + size_) { RETURN_NOT_OK(Resize(length + size_)); }
197  memset(data_ + size_, 0, static_cast<size_t>(length));
198  size_ += length;
199  return Status::OK();
200  }
