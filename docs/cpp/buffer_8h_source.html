181 class ARROW_EXPORT BufferBuilder {
182  public:
183  explicit BufferBuilder(MemoryPool* pool)
184  : pool_(pool), data_(nullptr), capacity_(0), size_(0) {}
185 
187  Status Resize(int64_t elements) {
188  // Resize(0) is a no-op
189  if (elements == 0) { return Status::OK(); }
190  if (capacity_ == 0) { buffer_ = std::make_shared<PoolBuffer>(pool_); }
191  int64_t old_capacity = capacity_;
192  RETURN_NOT_OK(buffer_->Resize(elements));
193  capacity_ = buffer_->capacity();
194  data_ = buffer_->mutable_data();
195  if (capacity_ > old_capacity) {
196  memset(data_ + old_capacity, 0, capacity_ - old_capacity);
197  }
198  return Status::OK();
199  }
200 
201  Status Append(const uint8_t* data, int64_t length) {
202  if (capacity_ < length + size_) { RETURN_NOT_OK(Resize(length + size_)); }
203  UnsafeAppend(data, length);
204  return Status::OK();
205  }
206 
207  // Advance pointer and zero out memory
208  Status Advance(int64_t length) {
209  if (capacity_ < length + size_) { RETURN_NOT_OK(Resize(length + size_)); }
210  memset(data_ + size_, 0, static_cast<size_t>(length));
211  size_ += length;
212  return Status::OK();
213  }
214 
215  // Unsafe methods don't check existing size
216  void UnsafeAppend(const uint8_t* data, int64_t length) {
217  memcpy(data_ + size_, data, static_cast<size_t>(length));
218  size_ += length;
219  }
220 
221  Status Finish(std::shared_ptr<Buffer>* out) {
222  // Do not shrink to fit to avoid unneeded realloc
223  if (size_ > 0) { RETURN_NOT_OK(buffer_->Resize(size_, false)); }
224  *out = buffer_;
225  Reset();
226  return Status::OK();
227  }
228 
229  void Reset() {
230  buffer_ = nullptr;
231  capacity_ = size_ = 0;
232  }
233 
234  int64_t capacity() const { return capacity_; }
235  int64_t length() const { return size_; }
236  const uint8_t* data() const { return data_; }
237 
238  protected:
239  std::shared_ptr<PoolBuffer> buffer_;
240  MemoryPool* pool_;
241  uint8_t* data_;
242  int64_t capacity_;
243  int64_t size_;
244 };
