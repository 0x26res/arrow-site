283 // Check if we would need to expand the underlying storage type
284 inline uint8_t expanded_uint_size(uint64_t val, uint8_t current_int_size) {
285  if (current_int_size == 8 ||
286  (current_int_size < 8 &&
287  (val > static_cast<uint64_t>(std::numeric_limits<uint32_t>::max())))) {
288  return 8;
289  } else if (current_int_size == 4 ||
290  (current_int_size < 4 &&
291  (val > static_cast<uint64_t>(std::numeric_limits<uint16_t>::max())))) {
292  return 4;
293  } else if (current_int_size == 2 ||
294  (current_int_size == 1 &&
295  (val > static_cast<uint64_t>(std::numeric_limits<uint8_t>::max())))) {
296  return 2;
297  } else {
298  return 1;
299  }
