297 // Check if we would need to expand the underlying storage type
298 inline uint8_t expanded_uint_size(uint64_t val, uint8_t current_int_size) {
299  if (current_int_size == 8 ||
300  (current_int_size < 8 &&
301  (val > static_cast<uint64_t>(std::numeric_limits<uint32_t>::max())))) {
302  return 8;
303  } else if (current_int_size == 4 ||
304  (current_int_size < 4 &&
305  (val > static_cast<uint64_t>(std::numeric_limits<uint16_t>::max())))) {
306  return 4;
307  } else if (current_int_size == 2 ||
308  (current_int_size == 1 &&
309  (val > static_cast<uint64_t>(std::numeric_limits<uint8_t>::max())))) {
310  return 2;
311  } else {
312  return 1;
313  }
