150  template <typename IterType>
151  void UnsafeAppendToBitmap(const IterType& begin, const IterType& end) {
152  int64_t byte_offset = length_ / 8;
153  int64_t bit_offset = length_ % 8;
154  uint8_t bitset = null_bitmap_data_[byte_offset];
155 
156  for (auto iter = begin; iter != end; ++iter) {
157  if (bit_offset == 8) {
158  bit_offset = 0;
159  null_bitmap_data_[byte_offset] = bitset;
160  byte_offset++;
161  // TODO: Except for the last byte, this shouldn't be needed
162  bitset = null_bitmap_data_[byte_offset];
163  }
164 
165  if (*iter) {
166  bitset |= BitUtil::kBitmask[bit_offset];
167  } else {
168  bitset &= BitUtil::kFlippedBitmask[bit_offset];
169  ++null_count_;
170  }
171 
172  bit_offset++;
173  }
174 
175  if (bit_offset != 0) {
176  null_bitmap_data_[byte_offset] = bitset;
177  }
178 
179  length_ += std::distance(begin, end);
180  }
181 
182  protected:
183  ArrayBuilder() {}
184

186  MemoryPool* pool_;
187

189  std::shared_ptr<ResizableBuffer> null_bitmap_;
190  int64_t null_count_;
191  uint8_t* null_bitmap_data_;

477 // Check if we would need to expand the underlying storage type
478 inline uint8_t ExpandedIntSize(int64_t val, uint8_t current_int_size) {
479  if (current_int_size == 8 ||
480  (current_int_size < 8 &&
481  (val > static_cast<int64_t>(std::numeric_limits<int32_t>::max()) ||
482  val < static_cast<int64_t>(std::numeric_limits<int32_t>::min())))) {
483  return 8;
484  } else if (current_int_size == 4 ||
485  (current_int_size < 4 &&
486  (val > static_cast<int64_t>(std::numeric_limits<int16_t>::max()) ||
487  val < static_cast<int64_t>(std::numeric_limits<int16_t>::min())))) {
488  return 4;
489  } else if (current_int_size == 2 ||
490  (current_int_size == 1 &&
491  (val > static_cast<int64_t>(std::numeric_limits<int8_t>::max()) ||
492  val < static_cast<int64_t>(std::numeric_limits<int8_t>::min())))) {
493  return 2;
494  } else {
495  return 1;
496  }
497 }
