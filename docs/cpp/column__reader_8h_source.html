106 class PARQUET_EXPORT ColumnReader {
107  public:
108  ColumnReader(const ColumnDescriptor*, std::unique_ptr<PageReader>,
109  ::arrow::MemoryPool* pool = ::arrow::default_memory_pool());
110  virtual ~ColumnReader();
111 
112  static std::shared_ptr<ColumnReader> Make(
113  const ColumnDescriptor* descr, std::unique_ptr<PageReader> pager,
114  ::arrow::MemoryPool* pool = ::arrow::default_memory_pool());
115 
116  // Returns true if there are still values in this column.
117  bool HasNext() {
118  // Either there is no data page available yet, or the data page has been
119  // exhausted
120  if (num_buffered_values_ == 0 || num_decoded_values_ == num_buffered_values_) {
121  if (!ReadNewPage() || num_buffered_values_ == 0) {
122  return false;
123  }
124  }
125  return true;
126  }
127 
128  Type::type type() const { return descr_->physical_type(); }
129 
130  const ColumnDescriptor* descr() const { return descr_; }
131 
132  protected:
133  virtual bool ReadNewPage() = 0;
134 
135  // Read multiple definition levels into preallocated memory
136  //
137  // Returns the number of decoded definition levels
138  int64_t ReadDefinitionLevels(int64_t batch_size, int16_t* levels);
139 
140  // Read multiple repetition levels into preallocated memory
141  // Returns the number of decoded repetition levels
142  int64_t ReadRepetitionLevels(int64_t batch_size, int16_t* levels);
143 
144  int64_t available_values_current_page() const {
145  return num_buffered_values_ - num_decoded_values_;
146  }
147 
148  void ConsumeBufferedValues(int64_t num_values) { num_decoded_values_ += num_values; }
149 
150  const ColumnDescriptor* descr_;
151 
152  std::unique_ptr<PageReader> pager_;
153  std::shared_ptr<Page> current_page_;
154 
155  // Not set if full schema for this field has no optional or repeated elements
156  LevelDecoder definition_level_decoder_;
157 
158  // Not set for flat schemas.
159  LevelDecoder repetition_level_decoder_;
160 
161  // The total number of values stored in the data page. This is the maximum of
162  // the number of encoded definition levels or encoded values. For
163  // non-repeated, required columns, this is equal to the number of encoded
164  // values. For repeated or optional values, there may be fewer data values
165  // than levels, and this tells you how many encoded levels there are in that

178 static inline void DefinitionLevelsToBitmap(
179  const int16_t* def_levels, int64_t num_def_levels, const int16_t max_definition_level,
180  const int16_t max_repetition_level, int64_t* values_read, int64_t* null_count,
181  uint8_t* valid_bits, const int64_t valid_bits_offset) {
182  ::arrow::internal::BitmapWriter valid_bits_writer(valid_bits, valid_bits_offset,
183  num_def_levels);
184 
185  // TODO(itaiin): As an interim solution we are splitting the code path here
186  // between repeated+flat column reads, and non-repeated+nested reads.
187  // Those paths need to be merged in the future
188  for (int i = 0; i < num_def_levels; ++i) {
189  if (def_levels[i] == max_definition_level) {
190  valid_bits_writer.Set();
191  } else if (max_repetition_level > 0) {
192  // repetition+flat case
193  if (def_levels[i] == (max_definition_level - 1)) {
194  valid_bits_writer.Clear();
195  *null_count += 1;
196  } else {
197  continue;
198  }
199  } else {
200  // non-repeated+nested case
201  if (def_levels[i] < max_definition_level) {
202  valid_bits_writer.Clear();
203  *null_count += 1;
204  } else {
205  throw ParquetException("definition level exceeds maximum");
206  }
207  }
208 
209  valid_bits_writer.Next();
210  }
211  valid_bits_writer.Finish();
212  *values_read = valid_bits_writer.position();
213 }

218 template <typename DType>
219 class PARQUET_TEMPLATE_CLASS_EXPORT TypedColumnReader : public ColumnReader {
220  public:
221  typedef typename DType::c_type T;
222 
223  TypedColumnReader(const ColumnDescriptor* schema, std::unique_ptr<PageReader> pager,
224  ::arrow::MemoryPool* pool = ::arrow::default_memory_pool())
225  : ColumnReader(schema, std::move(pager), pool), current_decoder_(NULLPTR) {}
226 
227  // Read a batch of repetition levels, definition levels, and values from the
228  // column.
229  //
230  // Since null values are not stored in the values, the number of values read
231  // may be less than the number of repetition and definition levels. With
232  // nested data this is almost certainly true.
233  //
234  // Set def_levels or rep_levels to nullptr if you want to skip reading them.
235  // This is only safe if you know through some other source that there are no
236  // undefined values.
237  //
238  // To fully exhaust a row group, you must read batches until the number of
239  // values read reaches the number of stored values according to the metadata.
240  //
241  // This API is the same for both V1 and V2 of the DataPage
242  //
243  // @returns: actual number of levels read (see values_read for number of values read)
244  int64_t ReadBatch(int64_t batch_size, int16_t* def_levels, int16_t* rep_levels,
245  T* values, int64_t* values_read);
246 
281  int64_t ReadBatchSpaced(int64_t batch_size, int16_t* def_levels, int16_t* rep_levels,
282  T* values, uint8_t* valid_bits, int64_t valid_bits_offset,
283  int64_t* levels_read, int64_t* values_read,
284  int64_t* null_count);
285 
286  // Skip reading levels
287  // Returns the number of levels skipped
288  int64_t Skip(int64_t num_rows_to_skip);
289 
290  private:
291  typedef Decoder<DType> DecoderType;
292 
293  // Advance to the next data page
294  bool ReadNewPage() override;
295 
296  // Read up to batch_size values from the current data page into the
297  // pre-allocated memory T*
298  //
299  // @returns: the number of values read into the out buffer
300  int64_t ReadValues(int64_t batch_size, T* out);
301 
302  // Read up to batch_size values from the current data page into the
303  // pre-allocated memory T*, leaving spaces for null entries according
304  // to the def_levels.
305  //
306  // @returns: the number of values read into the out buffer
307  int64_t ReadValuesSpaced(int64_t batch_size, T* out, int64_t null_count,
308  uint8_t* valid_bits, int64_t valid_bits_offset);
309 
310  // Map of encoding type to the respective decoder object. For example, a
311  // column chunk's data pages may include both dictionary-encoded and
312  // plain-encoded data.
313  std::unordered_map<int, std::shared_ptr<DecoderType>> decoders_;
314 
315  void ConfigureDictionary(const DictionaryPage* page);
316 
317  DecoderType* current_decoder_;
318 };

339 template <typename DType>
340 inline int64_t TypedColumnReader<DType>::ReadBatch(int64_t batch_size,
341  int16_t* def_levels,
342  int16_t* rep_levels, T* values,
343  int64_t* values_read) {
344  // HasNext invokes ReadNewPage
345  if (!HasNext()) {
346  *values_read = 0;
347  return 0;
348  }
349 
350  // TODO(wesm): keep reading data pages until batch_size is reached, or the
351  // row group is finished
352  batch_size = std::min(batch_size, num_buffered_values_ - num_decoded_values_);
353 
354  int64_t num_def_levels = 0;
355  int64_t num_rep_levels = 0;
356 
357  int64_t values_to_read = 0;
358 
359  // If the field is required and non-repeated, there are no definition levels
360  if (descr_->max_definition_level() > 0 && def_levels) {
361  num_def_levels = ReadDefinitionLevels(batch_size, def_levels);
362  // TODO(wesm): this tallying of values-to-decode can be performed with better
363  // cache-efficiency if fused with the level decoding.
364  for (int64_t i = 0; i < num_def_levels; ++i) {
365  if (def_levels[i] == descr_->max_definition_level()) {
366  ++values_to_read;
367  }
368  }
369  } else {
370  // Required field, read all values
371  values_to_read = batch_size;
372  }
373 
374  // Not present for non-repeated fields
375  if (descr_->max_repetition_level() > 0 && rep_levels) {
376  num_rep_levels = ReadRepetitionLevels(batch_size, rep_levels);
377  if (def_levels && num_def_levels != num_rep_levels) {
378  throw ParquetException("Number of decoded rep / def levels did not match");
379  }
380  }
381 
382  *values_read = ReadValues(values_to_read, values);
383  int64_t total_values = std::max(num_def_levels, *values_read);
384  ConsumeBufferedValues(total_values);
385 
386  return total_values;
387 }

391 // TODO(itaiin): another code path split to merge when the general case is done
392 static inline bool HasSpacedValues(const ColumnDescriptor* descr) {
393  if (descr->max_repetition_level() > 0) {
394  // repeated+flat case
395  return !descr->schema_node()->is_required();
396  } else {
397  // non-repeated+nested case
398  // Find if a node forces nulls in the lowest level along the hierarchy
399  const schema::Node* node = descr->schema_node().get();
400  while (node) {
401  if (node->is_optional()) {
402  return true;
403  }
404  node = node->parent();
405  }
406  return false;
407  }
408 }

412 template <typename DType>
413 inline int64_t TypedColumnReader<DType>::ReadBatchSpaced(
414  int64_t batch_size, int16_t* def_levels, int16_t* rep_levels, T* values,
415  uint8_t* valid_bits, int64_t valid_bits_offset, int64_t* levels_read,
416  int64_t* values_read, int64_t* null_count_out) {
417  // HasNext invokes ReadNewPage
418  if (!HasNext()) {
419  *levels_read = 0;
420  *values_read = 0;
421  *null_count_out = 0;
422  return 0;
423  }
424 
425  int64_t total_values;
426  // TODO(wesm): keep reading data pages until batch_size is reached, or the
427  // row group is finished
428  batch_size = std::min(batch_size, num_buffered_values_ - num_decoded_values_);
429 
430  // If the field is required and non-repeated, there are no definition levels
431  if (descr_->max_definition_level() > 0) {
432  int64_t num_def_levels = ReadDefinitionLevels(batch_size, def_levels);
433 
434  // Not present for non-repeated fields
435  if (descr_->max_repetition_level() > 0) {
436  int64_t num_rep_levels = ReadRepetitionLevels(batch_size, rep_levels);
437  if (num_def_levels != num_rep_levels) {
438  throw ParquetException("Number of decoded rep / def levels did not match");
439  }
440  }
441 
442  const bool has_spaced_values = internal::HasSpacedValues(descr_);
443 
444  int64_t null_count = 0;
445  if (!has_spaced_values) {
446  int values_to_read = 0;
447  for (int64_t i = 0; i < num_def_levels; ++i) {
448  if (def_levels[i] == descr_->max_definition_level()) {
449  ++values_to_read;
450  }
451  }
452  total_values = ReadValues(values_to_read, values);
453  for (int64_t i = 0; i < total_values; i++) {
454  ::arrow::BitUtil::SetBit(valid_bits, valid_bits_offset + i);
455  }
456  *values_read = total_values;
457  } else {
458  int16_t max_definition_level = descr_->max_definition_level();
459  int16_t max_repetition_level = descr_->max_repetition_level();
460  internal::DefinitionLevelsToBitmap(def_levels, num_def_levels, max_definition_level,
461  max_repetition_level, values_read, &null_count,
462  valid_bits, valid_bits_offset);
463  total_values = ReadValuesSpaced(*values_read, values, static_cast<int>(null_count),
464  valid_bits, valid_bits_offset);
465  }
466  *levels_read = num_def_levels;
467  *null_count_out = null_count;
468 
469  } else {
470  // Required field, read all values
471  total_values = ReadValues(batch_size, values);
472  for (int64_t i = 0; i < total_values; i++) {
473  ::arrow::BitUtil::SetBit(valid_bits, valid_bits_offset + i);
474  }
475  *null_count_out = 0;
476  *levels_read = total_values;
477  }
478 
479  ConsumeBufferedValues(*levels_read);
480  return total_values;
481 }

483 template <typename DType>
484 int64_t TypedColumnReader<DType>::Skip(int64_t num_rows_to_skip) {
485  int64_t rows_to_skip = num_rows_to_skip;
486  while (HasNext() && rows_to_skip > 0) {
487  // If the number of rows to skip is more than the number of undecoded values, skip the
488  // Page.
489  if (rows_to_skip > (num_buffered_values_ - num_decoded_values_)) {
490  rows_to_skip -= num_buffered_values_ - num_decoded_values_;
491  num_decoded_values_ = num_buffered_values_;
492  } else {
493  // We need to read this Page
494  // Jump to the right offset in the Page
495  int64_t batch_size = 1024; // ReadBatch with a smaller memory footprint
496  int64_t values_read = 0;
497 
498  std::shared_ptr<ResizableBuffer> vals = AllocateBuffer(
499  this->pool_, batch_size * type_traits<DType::type_num>::value_byte_size);
500  std::shared_ptr<ResizableBuffer> def_levels =
501  AllocateBuffer(this->pool_, batch_size * sizeof(int16_t));
502 
503  std::shared_ptr<ResizableBuffer> rep_levels =
504  AllocateBuffer(this->pool_, batch_size * sizeof(int16_t));
505 
506  do {
507  batch_size = std::min(batch_size, rows_to_skip);
508  values_read = ReadBatch(static_cast<int>(batch_size),
509  reinterpret_cast<int16_t*>(def_levels->mutable_data()),
510  reinterpret_cast<int16_t*>(rep_levels->mutable_data()),
511  reinterpret_cast<T*>(vals->mutable_data()), &values_read);
512  rows_to_skip -= values_read;
513  } while (values_read > 0 && rows_to_skip > 0);
514  }
515  }
516  return num_rows_to_skip - rows_to_skip;
517 }
