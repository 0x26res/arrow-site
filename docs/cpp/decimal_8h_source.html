87 template <typename T>
88 ARROW_EXPORT std::string ToString(
89  const Decimal<T>& decimal_value, int precision, int scale) {
90  T value = decimal_value.value;
91 
92  // Decimal values are sent to clients as strings so in the interest of
93  // speed the string will be created without the using stringstream with the
94  // whole/fractional_part().
95  size_t last_char_idx = precision + (scale > 0) // Add a space for decimal place
96  + (scale == precision) // Add a space for leading 0
97  + (value < 0); // Add a space for negative sign
98  std::string str = std::string(last_char_idx, '0');
99  // Start filling in the values in reverse order by taking the last digit
100  // of the value. Use a positive value and worry about the sign later. At this
101  // point the last_char_idx points to the string terminator.
102  T remaining_value = value;
103  size_t first_digit_idx = 0;
104  if (value < 0) {
105  remaining_value = -value;
106  first_digit_idx = 1;
107  }
108  if (scale > 0) {
109  int remaining_scale = scale;
110  do {
111  str[--last_char_idx] = static_cast<char>(
112  (remaining_value % 10) + static_cast<T>('0')); // Ascii offset
113  remaining_value /= 10;
114  } while (--remaining_scale > 0);
115  str[--last_char_idx] = '.';
116  DCHECK_GT(last_char_idx, first_digit_idx) << "Not enough space remaining";
117  }
118  do {
119  str[--last_char_idx] =
120  static_cast<char>((remaining_value % 10) + static_cast<T>('0')); // Ascii offset
121  remaining_value /= 10;
122  if (remaining_value == 0) {
123  // Trim any extra leading 0's.
124  if (last_char_idx > first_digit_idx) str.erase(0, last_char_idx - first_digit_idx);
125  break;
126  }
127  // For safety, enforce string length independent of remaining_value.
128  } while (last_char_idx > first_digit_idx);
129  if (value < 0) str[0] = '-';
130  return str;
131 }
