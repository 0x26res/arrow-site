44 static inline Status ZeroMemoryMap(MemoryMappedFile* file) {
45  constexpr int64_t kBufferSize = 512;
46  static constexpr uint8_t kZeroBytes[kBufferSize] = {0};
47 
48  RETURN_NOT_OK(file->Seek(0));
49  int64_t position = 0;
50  int64_t file_size;
51  RETURN_NOT_OK(file->GetSize(&file_size));
52 
53  int64_t chunksize;
54  while (position < file_size) {
55  chunksize = std::min(kBufferSize, file_size - position);
56  RETURN_NOT_OK(file->Write(kZeroBytes, chunksize));
57  position += chunksize;
58  }
59  return Status::OK();
60 }
