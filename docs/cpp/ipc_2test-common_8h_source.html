51 static inline void CompareBatch(const RecordBatch& left, const RecordBatch& right) {
52  if (!left.schema()->Equals(*right.schema())) {
53  FAIL() << "Left schema: " << left.schema()->ToString()
54  << "\nRight schema: " << right.schema()->ToString();
55  }
56  ASSERT_EQ(left.num_columns(), right.num_columns())
57  << left.schema()->ToString() << " result: " << right.schema()->ToString();
58  EXPECT_EQ(left.num_rows(), right.num_rows());
59  for (int i = 0; i < left.num_columns(); ++i) {
60  EXPECT_TRUE(left.column(i)->Equals(right.column(i)))
61  << "Idx: " << i << " Name: " << left.column_name(i);
62  }

105 Status MakeRandomListArray(const std::shared_ptr<Array>& child_array, int num_lists,
106  bool include_nulls, MemoryPool* pool, std::shared_ptr<Array>* out) {
107  // Create the null list values
108  std::vector<uint8_t> valid_lists(num_lists);
109  const double null_percent = include_nulls ? 0.1 : 0;
110  test::random_null_bytes(num_lists, null_percent, valid_lists.data());
111 
112  // Create list offsets
113  const int max_list_size = 10;
114 
115  std::vector<int32_t> list_sizes(num_lists, 0);
116  std::vector<int32_t> offsets(
117  num_lists + 1, 0); // +1 so we can shift for nulls. See partial sum below.
118  const uint32_t seed = static_cast<uint32_t>(child_array->length());
119  if (num_lists > 0) {
120  test::rand_uniform_int(num_lists, seed, 0, max_list_size, list_sizes.data());
121  // make sure sizes are consistent with null
122  std::transform(list_sizes.begin(), list_sizes.end(), valid_lists.begin(),
123  list_sizes.begin(),
124  [](int32_t size, int32_t valid) { return valid == 0 ? 0 : size; });
125  std::partial_sum(list_sizes.begin(), list_sizes.end(), ++offsets.begin());
126 
127  // Force invariants
128  const int32_t child_length = static_cast<int32_t>(child_array->length());
129  offsets[0] = 0;
130  std::replace_if(offsets.begin(), offsets.end(),
131  [child_length](int32_t offset) { return offset > child_length; }, child_length);
132  }
133  ListBuilder builder(pool, child_array);
134  RETURN_NOT_OK(builder.Append(offsets.data(), num_lists, valid_lists.data()));
135  RETURN_NOT_OK(builder.Finish(out));
136  return (*out)->Validate();
137 }

195 Status MakeRandomBinaryArray(
196  int64_t length, bool include_nulls, MemoryPool* pool, std::shared_ptr<Array>* out) {
197  const std::vector<std::string> values = {
198  "", "", "abc", "123", "efg", "456!@#!@#", "12312"};
199  Builder builder(pool);
200  const size_t values_len = values.size();
201  for (int64_t i = 0; i < length; ++i) {
202  int64_t values_index = i % values_len;
203  if (include_nulls && values_index == 0) {
204  RETURN_NOT_OK(builder.AppendNull());
205  } else {
206  const std::string& value = values[values_index];
207  RETURN_NOT_OK(builder.Append(reinterpret_cast<const RawType*>(value.data()),
208  static_cast<int32_t>(value.size())));
209  }
210  }
211  return builder.Finish(out);
212 }
