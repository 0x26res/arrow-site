30 // Visit the Python sequence, calling the given callable on each element.
31 // If the callable returns a non-OK status, iteration stops and the status is returned.
32 template <class UnaryFunction>
33 inline Status VisitSequence(PyObject* obj, UnaryFunction&& func) {
34  if (PyArray_Check(obj)) {
35  PyArrayObject* arr_obj = reinterpret_cast<PyArrayObject*>(obj);
36  if (PyArray_NDIM(arr_obj) != 1) {
37  return Status::Invalid("Only 1D arrays accepted");
38  }
39 
40  if (PyArray_DESCR(arr_obj)->type_num == NPY_OBJECT) {
41  // It's an array object, we can fetch object pointers directly
42  const Ndarray1DIndexer<PyObject*> objects(arr_obj);
43  for (int64_t i = 0; i < objects.size(); ++i) {
44  RETURN_NOT_OK(func(objects[i]));
45  }
46  return Status::OK();
47  }
48  // It's a non-object array, fall back on regular sequence access.
49  // (note PyArray_GETITEM() is slightly different: it returns standard
50  // Python types, not Numpy scalar types)
51  // This code path is inefficient: callers should implement dedicated
52  // logic for non-object arrays.
53  }
54  if (PySequence_Check(obj)) {
55  if (PyList_Check(obj) || PyTuple_Check(obj)) {
56  // Use fast item access
57  const Py_ssize_t size = PySequence_Fast_GET_SIZE(obj);
58  for (Py_ssize_t i = 0; i < size; ++i) {
59  PyObject* value = PySequence_Fast_GET_ITEM(obj, i);
60  RETURN_NOT_OK(func(value));
61  }
62  } else {
63  // Regular sequence: avoid making a potentially large copy
64  const Py_ssize_t size = PySequence_Size(obj);
65  RETURN_IF_PYERROR();
66  for (Py_ssize_t i = 0; i < size; ++i) {
67  OwnedRef value_ref(PySequence_ITEM(obj, i));
68  RETURN_IF_PYERROR();
69  RETURN_NOT_OK(func(value_ref.obj()));
70  }
71  }
72  } else {
73  return Status::TypeError("Object is not a sequence");
74  }
75  return Status::OK();
76 }

78 // Like IterateSequence, but accepts any generic iterable (including
79 // non-restartable iterators, e.g. generators).
80 template <class UnaryFunction>
81 inline Status VisitIterable(PyObject* obj, UnaryFunction&& func) {
82  if (PySequence_Check(obj)) {
83  // Numpy arrays fall here as well
84  return VisitSequence(obj, std::forward<UnaryFunction>(func));
85  }
86  // Fall back on the iterator protocol
87  OwnedRef iter_ref(PyObject_GetIter(obj));
88  PyObject* iter = iter_ref.obj();
89  RETURN_IF_PYERROR();
90  PyObject* value;
91  while ((value = PyIter_Next(iter))) {
92  OwnedRef value_ref(value);
93  RETURN_NOT_OK(func(value_ref.obj()));
94  }
95  RETURN_IF_PYERROR(); // __next__() might have raised
96  return Status::OK();
97 }
