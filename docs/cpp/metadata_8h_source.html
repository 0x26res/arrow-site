40 class ApplicationVersion {
41  public:
42  // Known Versions with Issues
43  static const ApplicationVersion& PARQUET_251_FIXED_VERSION();
44  static const ApplicationVersion& PARQUET_816_FIXED_VERSION();
45  static const ApplicationVersion& PARQUET_CPP_FIXED_STATS_VERSION();
46  // Regular expression for the version format
47  // major . minor . patch unknown - prerelease.x + build info
48  // Eg: 1.5.0ab-cdh5.5.0+cd
49  static constexpr char const* VERSION_FORMAT =
50  "^(\\d+)\\.(\\d+)\\.(\\d+)([^-+]*)?(?:-([^+]*))?(?:\\+(.*))?$";
51  // Regular expression for the application format
52  // application_name version VERSION_FORMAT (build build_name)
53  // Eg: parquet-cpp version 1.5.0ab-xyz5.5.0+cd (build abcd)
54  static constexpr char const* APPLICATION_FORMAT =
55  "(.*?)\\s*(?:(version\\s*(?:([^(]*?)\\s*(?:\\(\\s*build\\s*([^)]*?)\\s*\\))?)?)?)";
56 
57  // Application that wrote the file. e.g. "IMPALA"
58  std::string application_;
59  // Build name
60  std::string build_;
61 
62  // Version of the application that wrote the file, expressed as
63  // (<major>.<minor>.<patch>). Unmatched parts default to 0.
64  // "1.2.3" => {1, 2, 3}
65  // "1.2" => {0, 0, 0}
66  // "1.2-cdh5" => {0, 0, 0}
67  // TODO (majetideepak): Implement support for pre_release
68  struct {
69  int major;
70  int minor;
71  int patch;
72  std::string unknown;
73  std::string pre_release;
74  std::string build_info;
75  } version;
76 
77  ApplicationVersion() {}
78  explicit ApplicationVersion(const std::string& created_by);
79  ApplicationVersion(const std::string& application, int major, int minor, int patch);
80 
81  // Returns true if version is strictly less than other_version
82  bool VersionLt(const ApplicationVersion& other_version) const;
83 
84  // Returns true if version is strictly less than other_version
85  bool VersionEq(const ApplicationVersion& other_version) const;
86 
87  // Checks if the Version has the correct statistics for a given column
88  bool HasCorrectStatistics(Type::type primitive, EncodedStatistics& statistics,
89  SortOrder::type sort_order = SortOrder::SIGNED) const;
90 };
