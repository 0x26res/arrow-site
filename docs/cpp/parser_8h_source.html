77  template <typename Visitor>
78  Status VisitColumn(int32_t col_index, Visitor&& visit) const {
79 #ifdef CSV_PARSER_USE_BITFIELD
80  for (int32_t pos = col_index; pos < num_cols_ * num_rows_; pos += num_cols_) {
81  auto start = values_[pos].offset;
82  auto stop = values_[pos + 1].offset;
83  auto quoted = values_[pos + 1].quoted;
84  ARROW_RETURN_NOT_OK(visit(parsed_.data() + start, stop - start, quoted));
85  }
86 #else
87  for (int32_t pos = col_index; pos < num_cols_ * num_rows_; pos += num_cols_) {
88  auto start = offsets_[pos];
89  auto stop = offsets_[pos + 1];
90  auto quoted = quoted_[pos];
91  ARROW_RETURN_NOT_OK(visit(parsed_.data() + start, stop - start, quoted));
92  }
93 #endif
94  return Status::OK();
95  }
