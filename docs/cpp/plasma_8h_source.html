38 #define HANDLE_SIGPIPE(s, fd_) \
39  do { \
40  Status _s = (s); \
41  if (!_s.ok()) { \
42  if (errno == EPIPE || errno == EBADF || errno == ECONNRESET) { \
43  ARROW_LOG(WARNING) \
44  << "Received SIGPIPE, BAD FILE DESCRIPTOR, or ECONNRESET when " \
45  "sending a message to client on fd " \
46  << fd_ << ". " \
47  "The client on the other end may have hung up."; \
48  } else { \
49  return _s; \
50  } \
51  } \
52  } while (0);
