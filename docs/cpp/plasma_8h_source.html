40 #define HANDLE_SIGPIPE(s, fd_) \
41  do { \
42  Status _s = (s); \
43  if (!_s.ok()) { \
44  if (errno == EPIPE || errno == EBADF || errno == ECONNRESET) { \
45  ARROW_LOG(WARNING) \
46  << "Received SIGPIPE, BAD FILE DESCRIPTOR, or ECONNRESET when " \
47  "sending a message to client on fd " \
48  << fd_ << ". " \
49  "The client on the other end may have hung up."; \
50  } else { \
51  return _s; \
52  } \
53  } \
54  } while (0);
