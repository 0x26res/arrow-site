230 static inline int numpy_type_size(int npy_type) {
231  switch (npy_type) {
232  case NPY_BOOL:
233  return 1;
234  case NPY_INT8:
235  return 1;
236  case NPY_INT16:
237  return 2;
238  case NPY_INT32:
239  return 4;
240  case NPY_INT64:
241  return 8;
242 #if (NPY_INT64 != NPY_LONGLONG)
243  case NPY_LONGLONG:
244  return 8;
245 #endif
246  case NPY_UINT8:
247  return 1;
248  case NPY_UINT16:
249  return 2;
250  case NPY_UINT32:
251  return 4;
252  case NPY_UINT64:
253  return 8;
254 #if (NPY_UINT64 != NPY_ULONGLONG)
255  case NPY_ULONGLONG:
256  return 8;
257 #endif
258  case NPY_FLOAT16:
259  return 2;
260  case NPY_FLOAT32:
261  return 4;
262  case NPY_FLOAT64:
263  return 8;
264  case NPY_DATETIME:
265  return 8;
266  case NPY_OBJECT:
267  return sizeof(void*);
268  default:
269  DCHECK(false) << "unhandled numpy type";
270  break;
271  }
272  return -1;
