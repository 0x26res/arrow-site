23 // A very simple random number generator. Not especially good at
24 // generating truly random bits, but good enough for our needs in this
25 // package. This implementation is not thread-safe.
26 class Random {
27  public:
28  explicit Random(uint32_t s) : seed_(s & 0x7fffffffu) {
29  // Avoid bad seeds.
30  if (seed_ == 0 || seed_ == random_internal::M) { seed_ = 1; }
31  }
32 
33  // Next pseudo-random 32-bit unsigned integer.
34  // FIXME: This currently only generates 31 bits of randomness.
35  // The MSB will always be zero.
36  uint32_t Next() {
37  static const uint64_t A = 16807; // bits 14, 8, 7, 5, 2, 1, 0
38  // We are computing
39  // seed_ = (seed_ * A) % M, where M = 2^31-1
40  //
41  // seed_ must not be zero or M, or else all subsequent computed values
42  // will be zero or M respectively. For all other values, seed_ will end
43  // up cycling through every number in [1,M-1]
44  uint64_t product = seed_ * A;
45 
46  // Compute (product % M) using the fact that ((x << 31) % M) == x.
47  seed_ = static_cast<uint32_t>((product >> 31) + (product & random_internal::M));
48  // The first reduction may overflow by 1 bit, so we may need to
49  // repeat. mod == M is not possible; using > allows the faster
50  // sign-bit-based test.
51  if (seed_ > random_internal::M) { seed_ -= random_internal::M; }
52  return seed_;
53  }
54 
55  // Alias for consistency with Next64
56  uint32_t Next32() { return Next(); }
57 
58  // Next pseudo-random 64-bit unsigned integer.
59  // FIXME: This currently only generates 62 bits of randomness due to Next()
60  // only giving 31 bits of randomness. The 2 most significant bits will always
61  // be zero.
62  uint64_t Next64() {
63  uint64_t large = Next();
64  // Only shift by 31 bits so we end up with zeros in MSB and not scattered
65  // throughout the 64-bit word. This is due to the weakness in Next() noted
66  // above.
67  large <<= 31;
68  large |= Next();
69  return large;
70  }
71 
72  // Returns a uniformly distributed value in the range [0..n-1]
73  // REQUIRES: n > 0
74  uint32_t Uniform(uint32_t n) { return Next() % n; }
75 
76  // Alias for consistency with Uniform64
77  uint32_t Uniform32(uint32_t n) { return Uniform(n); }
78 
79  // Returns a uniformly distributed 64-bit value in the range [0..n-1]
80  // REQUIRES: n > 0
81  uint64_t Uniform64(uint64_t n) { return Next64() % n; }
82 
83  // Randomly returns true ~"1/n" of the time, and false otherwise.
84  // REQUIRES: n > 0
85  bool OneIn(int n) { return (Next() % n) == 0; }
86 
87  // Skewed: pick "base" uniformly from range [0,max_log] and then
88  // return "base" random bits. The effect is to pick a number in the
89  // range [0,2^max_log-1] with exponential bias towards smaller numbers.
90  uint32_t Skewed(int max_log) { return Uniform(1 << Uniform(max_log + 1)); }
91 
92  // Creates a normal distribution variable using the
93  // Box-Muller transform. See:
94  // http://en.wikipedia.org/wiki/Box%E2%80%93Muller_transform
95  // Adapted from WebRTC source code at:
96  // webrtc/trunk/modules/video_coding/main/test/test_util.cc
97  double Normal(double mean, double std_dev) {
98  double uniform1 = (Next() + 1.0) / (random_internal::M + 1.0);
99  double uniform2 = (Next() + 1.0) / (random_internal::M + 1.0);
100  return (
101  mean +
102  std_dev * sqrt(-2 * ::log(uniform1)) * cos(random_internal::kTwoPi * uniform2));
103  }
104 
105  // Return a random number between 0.0 and 1.0 inclusive.
106  double NextDoubleFraction() {
107  return Next() / static_cast<double>(random_internal::M + 1.0);
108  }
109 
110  private:
111  uint32_t seed_;
112 };
