136 class ARROW_EXPORT DataType {
137  public:
138  explicit DataType(Type::type id) : id_(id) {}
139  virtual ~DataType();

142  //
143  // Types that are logically convertable from one to another e.g. List<UInt8>
144  // and Binary are NOT equal).
145  virtual bool Equals(const DataType& other) const;

147 
148  std::shared_ptr<Field> child(int i) const { return children_[i]; }
149 
150  const std::vector<std::shared_ptr<Field>>& children() const { return children_; }
151 
152  int num_children() const { return static_cast<int>(children_.size()); }
153

155 
157  virtual std::string ToString() const = 0;
158 
163  virtual std::string name() const = 0;
164 
165  Type::type id() const { return id_; }
166 
167  protected:
168  Type::type id_;

170 
171  private:
172  ARROW_DISALLOW_COPY_AND_ASSIGN(DataType);
173 };

586  case TimeUnit::MILLI:

588  break;
589  case TimeUnit::MICRO:

591  break;
592  case TimeUnit::NANO:
593  os << "ns";
594  break;
595  }
596  return os;
