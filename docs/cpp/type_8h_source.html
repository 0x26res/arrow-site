38 // Data types in this library are all *logical*. They can be expressed as
39 // either a primitive physical type (bytes or bits of some fixed size), a
40 // nested type consisting of other data types, or another data type (e.g. a
41 // timestamp encoded as an int64)
42 struct Type {
43  enum type {
44  // A degenerate NULL type represented as 0 bytes/bits
45  NA,
46 
47  // A boolean value represented as 1 bit
48  BOOL,
49 
50  // Little-endian integer types
51  UINT8,
52  INT8,
53  UINT16,
54  INT16,
55  UINT32,
56  INT32,
57  UINT64,
58  INT64,
59 
60  // 2-byte floating point value
61  HALF_FLOAT,
62 
63  // 4-byte floating point value
64  FLOAT,
65 
66  // 8-byte floating point value
67  DOUBLE,
68 
69  // UTF8 variable-length string as List<Char>
70  STRING,
71 
72  // Variable-length bytes (no guarantee of UTF8-ness)
73  BINARY,
74 
75  // Fixed-size binary. Each value occupies the same number of bytes
76  FIXED_SIZE_BINARY,
77 
78  // int32_t days since the UNIX epoch
79  DATE32,
80 
81  // int64_t milliseconds since the UNIX epoch
82  DATE64,
83 
84  // Exact timestamp encoded with int64 since UNIX epoch
85  // Default unit millisecond
86  TIMESTAMP,
87 
88  // Time as signed 32-bit integer, representing either seconds or
89  // milliseconds since midnight
90  TIME32,
91 
92  // Time as signed 64-bit integer, representing either microseconds or
93  // nanoseconds since midnight
94  TIME64,
95 
96  // YEAR_MONTH or DAY_TIME interval in SQL style
97  INTERVAL,
98 
99  // Precision- and scale-based decimal type. Storage type depends on the
100  // parameters.
101  DECIMAL,
102 
103  // A list of some logical data type
104  LIST,
105 
106  // Struct of logical types
107  STRUCT,
108 
109  // Unions of logical types
110  UNION,
111 
112  // Dictionary aka Category type
113  DICTIONARY
114  };
115 };

447 static inline int decimal_byte_width(int precision) {
448  if (precision >= 0 && precision < 10) {
449  return 4;

451  return 8;
