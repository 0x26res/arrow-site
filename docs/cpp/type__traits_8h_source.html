306 #define GET_ATTR(ATTR_NAME, DEFAULT) \
307  template <typename T, typename Enable = void> \
308  struct GetAttr_##ATTR_NAME { \
309  using type = DEFAULT; \
310  }; \
311  \
312  template <typename T> \
313  struct GetAttr_##ATTR_NAME<T, typename as_void<typename T::ATTR_NAME>::type> { \
314  using type = typename T::ATTR_NAME; \
315  };
